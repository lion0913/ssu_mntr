#include<stdio.h>
#include<stdlib.h>

int main(void){
	char *fname="ssu_test.txt";
	FILE *fp;
	long fsize;
	if((fp=fopen(fname,"r"))==NULL){//파일오픈(읽기권한)
		fprintf(stderr,"fopen error for %s\n",fname);
		exit(1);
	}//파일오픈 에러처리
	fseek(fp,0,SEEK_END);//파일끝으로 오프셋 이동
	fsize=ftell(fp);//파일의 크기 fsize에 저장
	printf("The size of <%s> is %ld bytes\n",fname,fsize);//출력
	exit(0);
}
